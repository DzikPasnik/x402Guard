```rust
//! Rate limiting middleware backed by Redis.
//!
//! # Phase 0 Stub
//!
//! This module provides:
//! - Redis connection validation at startup
//! - `GlobalRateLimitConfig` for configuration
//!
//! # Phase 1 Implementation
//!
//! TODO: Implement sliding window rate limiter using:
//! - `ZADD key timestamp timestamp` — add request timestamp
//! - `ZREMRANGEBYSCORE key 0 (now - window)` — evict old entries
//! - `ZCARD key` — count requests in window
//! - `SET NX EX` for nonce deduplication
//! - Return 429 with `Retry-After` header on exceed (FR-5.4)

use anyhow::Context;
use redis::Client;

/// Global rate limit configuration (FR-5.3).
///
/// Controls the maximum number of requests per second accepted
/// by the proxy across all agents. Provides DDoS protection.
#[derive(Debug, Clone)]
pub struct GlobalRateLimitConfig {
    /// Maximum requests per second allowed globally.
    pub requests_per_second: u32,
}

impl GlobalRateLimitConfig {
    pub fn new(requests_per_second: u32) -> Self {
        Self { requests_per_second }
    }
}

/// Create and validate a Redis client from a connection URL.
///
/// Validates connectivity at startup so misconfigured Redis
/// is caught immediately rather than at first request.
///
/// # Errors
///
/// Returns an error if the URL is malformed or the connection
/// cannot be established.
pub async fn create_redis_client(url: &str) -> anyhow::Result<Client> {
    let client = Client::open(url)
        .context("failed to parse Redis URL")?;

    // Validate connection is reachable at startup.
    let mut conn = client
        .get_multiplexed_tokio_connection()
        .await
        .context("failed to connect to Redis")?;

    redis::cmd("PING")
        .query_async::<String>(&mut conn)
        .await
        .context("Redis PING failed")?;

    Ok(client)
}
```
